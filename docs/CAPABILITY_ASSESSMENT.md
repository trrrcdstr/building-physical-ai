```python
# scripts/parse_dwg.py
import ezdxf
import json

def parse_dwg_to_json(dwg_path):
    doc = ezdxf.readfile(dwg_path)
    msp = doc.modelspace()

    geometry = {
        'walls': [],
        'doors': [],
        'windows': [],
        'annotations': []
    }

    for entity in msp:
        if entity.dxftype() == 'LINE':
            # 提取线段
            start = entity.dxf.start
            end = entity.dxf.end
            geometry['walls'].append({
                'start': [start.x, start.y, start.z],
                'end': [end.x, end.y, end.z],
                'layer': entity.layer
            })

    return geometry
```
